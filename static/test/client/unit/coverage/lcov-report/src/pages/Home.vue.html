        },
 
        methods: {
            togglePlayVideo: function(e) {
                e.preventDefault();
                let $vid = this.$refs['video-element']
 
                if ($vid.paused) {
                  $vid.play();
                  this.video.playButtonText = this.$t('Pause');
                }
                else {
                  $vid.pause();
                  this.video.playButtonText = this.$t('Play');
                }
            },
 
            toggleMuteVideo: function(e) {
                e.preventDefault();
                let $vid = this.$refs['video-element']
 
                $vid.muted = $vid.muted ? false : true;
